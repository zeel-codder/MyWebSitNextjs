## P.1.Single Inheritance.

*Program:*

```python
# inheritance
#  "basically inheritance is menses we use properties of one class in
 #another class and we can mack code less and very helpful"
# single inheritance(P-->C)


class parent():
    def __init__(self,name,fname):
        self.name=name
        self.fname=fname
    def print(self):
        print('Name is',self.name,'Father name is',self.fname)
    def sum(self,a,b):
        self.a=a
        self.b=b
        return self.a+self.b
class child(parent):
    def __init__(self,name,fname):
        # super().__init__(name,fname)
        parent.__init__(self,name,fname)
    def display(self,sum):
        self.sum=sum
        print('sum is',sum)
        print('sum is',self.sum)
bo=child('zeel','Suresbhai')
# bo.name="veels"
bo.print()
bo.display(bo.sum(1,3))
```

## P.6.SUPER FUNCTION.

*Program:*

```python
class parent():
    def __init__(self,id,name):
        self.id=id
        self.name=name
    def print(self):
        print('I am super class')
    def display(self):
        print('Id is',self.id,'Name is',self.name)
class child(parent):
    def __init__(self,id,name): 
        super().__init__(id,name)#super() is say only this is main parent function
```
